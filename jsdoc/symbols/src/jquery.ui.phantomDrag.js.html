 23 /**
 24 * @class
 25 * @name phantomDrag
 26 * @param {object} options 設定オブジェクト
 27 * @config {number} xmax X座標の最大値
 28 * @config {number} xmin X座標の最小値
 29 * @config {number} ymax Y座標の最大値
 30 * @config {number} ymin Y座標の最小値
 31 * @config {number} delay ドラッグ追従の遅延具合。大きいほど遅く、0で遅延無し
 32 * @memberOf jQuery.ui
 33 * @description 要素をドラッグで移動したと仮定した値を jQuery.data に書き出します。<br>
 34 * ドラッグ中の各イベントに bind して、その値を流用することで様々な動きを実現します。<br><br>
 35 *
 36 * [Custom Event]<br>
 37 * phantomdrag-start: ドラッグ開始イベント<br>
 38 * phantomdrag-move: ドラッグ中イベント（ドラッグ中に毎フレーム呼ばれます）<br>
 39 * phantomdrag-release: ドラッグ解除イベント<br>
 40 * phantomdrag-stop: ドラッグ移動停止イベント（ドラッグ解除後も続く慣性移動が停止した合図です）<br><br>
 41 *
 42 * [jQuery.data]<br>
 43 * phantomdrag-x: 要素の現在のX座標<br>
 44 * phantomdrag-fx: 要素のドラッグ開始X座標<br>
 45 * phantomdrag-dx: 要素の移動先X座標<br>
 46 * phantomdrag-sx: 要素の1フレーム間でのX方向移動量<br>
 47 * phantomdrag-y: 要素の現在のY座標<br>
 48 * phantomdrag-fy: 要素のドラッグ開始Y座標<br>
 49 * phantomdrag-dy: 要素の移動先Y座標<br>
 50 * phantomdrag-sy: 要素の1フレーム間でのY方向移動量<br>
 51 *
 52 * @example
 53 * // 初期化
 54 *
 55 * $('#drag').phantomDrag({
 56 *     xmax: 660,
 57 *     xmin: 0,
 58 *     ymax: 260,
 59 *     ymin: 0
 60 * });
 61 *
 62 * // Custom Event への bind
 63 * // ドラッグ中の現在位置をCSSに反映する例です
 64 *
 65 * $('#drag').bind('phantomdrag-move', function (e) {
 66 *
 67 *     var $el = $(e.target);
 68 *
 69 *     $el.css({
 70 *         top: $el.data('phantomdrag-y'),
 71 *         left: $el.data('phantomdrag-x')
 72 *     })
 73 * });
 74 *
 75 */
 76 (function ($, window, document, Math, interval) {
 77 
 78     $.widget('ui.phantomDrag', {
 79 
 80         options: {
 81             xmax: 65536,
 82             xmin: -65536,
 83             ymax: 65536,
 84             ymin: -65536,
 85             delay: 13,
 86             customEasing: null,
 87             stopSpeedBorder: 0.1
 88         },
 89 
 90         EVENT_DRAG_START: 'phantomdrag-start',
 91         EVENT_DRAG_RELEASE: 'phantomdrag-release',
 92         EVENT_DRAG_MOVE: 'phantomdrag-move',
 93         EVENT_DRAG_STOP: 'phantomdrag-stop',
 94 
 95         mx: 0,
 96         mfx: 0,
 97         my: 0,
 98         mfy: 0,
 99         timer: 0,
100 
101         _init: function () {
102             this.element
103             .bind({
104                 'mousedown': $.proxy(this._start, this)
105             })
106             .data('phantomdrag-x', 0)
107             .data('phantomdrag-fx', 0)
108             .data('phantomdrag-dx', 0)
109             .data('phantomdrag-sx', 0)
110             .data('phantomdrag-y', 0)
111             .data('phantomdrag-fy', 0)
112             .data('phantomdrag-dy', 0)
113             .data('phantomdrag-sy', 0);
114         },
115 
116         _startTimer: function () {
117             this.timer = interval($.proxy(this._onEnterFrame, this), 16);
118         },
119 
120         _stopTimer: function () {
121             clearInterval(this.timer);
122             this.timer = 0;
123         },
124 
125         _start: function (e) {
126 
127             this._stopTimer();
128 
129             var x = this.element.data('phantomdrag-x'),
130                 y = this.element.data('phantomdrag-y');
131 
132             this.element
133             .data('phantomdrag-on', true)
134             .data('phantomdrag-dx', x)
135             .data('phantomdrag-fx', x)
136             .data('phantomdrag-dy', y)
137             .data('phantomdrag-fy', y);
138 
139             this.mfx = Math.round(e.pageX);
140             this.mfy = Math.round(e.pageY);
141 
142             this.element.trigger(this.EVENT_DRAG_START);
143 
144             $(document).bind({
145                 'mousemove': $.proxy(this._move, this),
146                 'mouseup': $.proxy(this._end, this)
147             });
148 
149             this._startTimer();
150 
151             return false;
152         },
153 
154         _move: function (e) {
155 
156             var o = this.options,
157                 $el = this.element;
158 
159             if (!$el.data('phantomdrag-move')) {
160                 $el.data('phantomdrag-move', true);
161             }
162 
163             this.mx = Math.round(e.pageX);
164             this.my = Math.round(e.pageY);
165 
166             $el
167             .data('phantomdrag-dx', Math.max(o.xmin, Math.min($el.data('phantomdrag-fx') + this.mx - this.mfx, o.xmax)))
168             .data('phantomdrag-dy', Math.max(o.ymin, Math.min($el.data('phantomdrag-fy') + this.my - this.mfy, o.ymax)));
169 
170             return false;
171         },
172 
173         _end: function (e) {
174 
175             this.element
176             .data('phantomdrag-on', false)
177             .trigger(this.EVENT_DRAG_RELEASE);
178 
179             $(document)
180             .unbind({
181                 'mousemove': this._move,
182                 'mouseup': this._end
183             });
184         },
185 
186         _onEnterFrame: function (e) {
187 
188             var $el = this.element,
189                 o = this.options,
190                 fx = $el.data('phantomdrag-fx'),
191                 sx = $el.data('phantomdrag-sx'),
192                 dx = $el.data('phantomdrag-dx'),
193                 x = $el.data('phantomdrag-x'),
194                 fy = $el.data('phantomdrag-fy'),
195                 sy = $el.data('phantomdrag-sy'),
196                 dy = $el.data('phantomdrag-dy'),
197                 y = $el.data('phantomdrag-y'),
198                 stop = null,
199                 border = o.stopSpeedBorder;
200 
201             sx = o.customEasing ? o.customEasing(x, dx, sx, fx) : (dx - x) / o.delay;
202             sy = o.customEasing ? o.customEasing(y, dy, sy, fy) : (dy - y) / o.delay;
203 
204             x += sx;
205             y += sy;
206 
207             stop = Math.abs(sx) < border && Math.abs(sy) < border;
208 
209             if (stop) {
210                 sx = 0;
211                 sy = 0;
212                 x = dx;
213                 y = dy;
214             }
215 
216             $el
217             .data('phantomdrag-x', x)
218             .data('phantomdrag-sx', sx)
219             .data('phantomdrag-dx', dx)
220             .data('phantomdrag-y', y)
221             .data('phantomdrag-sy', sy)
222             .data('phantomdrag-dy', dy);
223 
224             this.element.trigger(this.EVENT_DRAG_MOVE);
225 
226             if (stop) {
227                 this._stop();
228             }
229         },
230 
231         _stop: function () {
232 
233             if (!this.element.data('phantomdrag-on')) {
234 
235                 this._stopTimer();
236 
237                 this.element.trigger(this.EVENT_DRAG_STOP);
238 
239                 this.element.data('phantomdrag-move', false);
240             }
241             return false;
242         }
243 
244     });
245 })(jQuery, this, this.document, Math, setInterval);
